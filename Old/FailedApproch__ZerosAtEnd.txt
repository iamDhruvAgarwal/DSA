class Solution {
  public:
    void pushZerosToEnd(vector<int>& arr) {
        // code here
        //sort(arr.begin(),arr.end(), greater<int>());
        for(long long int i = 0; i < arr.size(); i++){
            for(long long int j = 0; j < arr.size(); j++){
                if(arr[j] == 0){
                    swap(arr[j],arr[j + 1]);
                }
            }
        }
    }
};
